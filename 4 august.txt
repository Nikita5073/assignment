{
	private:
		string name;
		int roll_no;
		int sub1_marks;
		int sub2_marks;
		int sub3_marks;
	public:
	student(string n,int r,int s1, int s2, int s3)
	{
		name=n;
		roll_no=r;
		sub1_marks=s1;
		sub2_marks=s2;
		sub3_marks=s3;
	}
	void getdetails()
	{
		cout<<"\n"<<"name is: "<<name;
		cout<<"\n"<<"roll number is: "<<roll_no;
	}
	void avgmarks()
	{
		int sum=sub1_marks+sub2_marks+sub3_marks;
		cout<<"\n"<<"avg is: "<<sum/3;
	}	
};
int main()
{
	student s2("vishal",20,80,85,75);
	s2.getdetails();
	s2.avgmarks();
}
